 227    def cook_image(self, img):
 228        """Converts an img to range [0, 1] for display.
 229
 230        The input image is stored in a `numpy.ndarray`, if it's dtype
 231        is `int` it will be rescaled and mapped into range [0, 1]. If
 232        the dtype is `float` it will be directly casted to 32-bit float type.
 233        """
 234        if img.dtype in [np.uint8, np.uint16, np.uint32, np.uint64]:
 235            img = img.astype(np.float32) * (1 / np.iinfo(img.dtype).max)
 236        elif img.dtype in [np.float16, np.float32, np.float64]:
 237            img = img.astype(np.float32)
 238        else:
 239            raise ValueError(f"Data type {img.dtype} not supported in GUI.set_image")
 240
 241        if len(img.shape) == 2:
 242            img = img[..., None]
 243
 244        if img.shape[2] == 1:
 245            img = img + np.zeros((1, 1, 4), np.float32)
 246        if img.shape[2] == 3:
 247            zeros = np.zeros((img.shape[0], img.shape[1], 1), np.float32)
 248            img = np.concatenate([img, zeros], axis=2)
 249        if img.shape[2] == 2:
 250            zeros = np.zeros((img.shape[0], img.shape[1], 2), np.float32)
 251            img = np.concatenate([img, zeros], axis=2)
 252
 253        assert img.shape[2] == 4, "Image must be grayscale, RG, RGB or RGBA"
 254
 255        res = img.shape[:2]
 256        assert res == self.res, "Image resolution does not match GUI resolution"
 257        return np.ascontiguousarray(img)

 269    def set_image(self, img):
 270        """Sets an image to display on the window.
 271
 272        The image pixels are set from the values of `img[i, j]`, where `i` indicates
 273        the horizontal coordinates (from left to right) and `j` the vertical coordinates
 274        (from bottom to top).
 275
 276        If the window size is `(x, y)`, then `img` must be one of:
 277            - `ti.field(shape=(x, y))`, a gray-scale image
 278            - `ti.field(shape=(x, y, 3))`, where `3` is for `(r, g, b)` channels
 279            - `ti.field(shape=(x, y, 2))`, where `2` is for `(r, g)` channels
 280            - `ti.Vector.field(3, shape=(x, y))` `(r, g, b)` channels on each component
 281            - `ti.Vector.field(2, shape=(x, y))` `(r, g)` channels on each component
 282            - `np.ndarray(shape=(x, y))`
 283            - `np.ndarray(shape=(x, y, 3))`
 284            - `np.ndarray(shape=(x, y, 2))`
 285
 286        The data type of `img` must be one of:
 287            - `uint8`, range `[0, 255]`
 288            - `uint16`, range `[0, 65535]`
 289            - `uint32`, range `[0, 4294967295]`
 290            - `float32`, range `[0, 1]`
 291            - `float64`, range `[0, 1]`
 292
 293        Args:
 294            img (Union[:class:`taichi.field`, `numpy.array`]): The color array \
 295                representing the image to be drawn. Support greyscale, RG, RGB, \
 296                and RGBA color representations. Its shape must match GUI resolution.
 297        """
 298
 299        if self.fast_gui:
 300            assert isinstance(
 301                img, taichi.lang.matrix.MatrixField
 302            ), "Only ti.Vector.field is supported in GUI.set_image when fast_gui=True"
 303            assert img.shape == self.res, "Image resolution does not match GUI resolution"
 304            assert img.n in [3, 4] and img.m == 1, "Only RGB images are supported in GUI.set_image when fast_gui=True"
 305            assert img.dtype in [
 306                ti.f32,
 307                ti.f64,
 308                ti.u8,
 309            ], "Only f32, f64, u8 are supported in GUI.set_image when fast_gui=True"
 310
 311            vector_to_fast_image(img, self.img)
 312            return
 313
 314        if isinstance(img, ScalarField):
 315            if _ti_core.is_integral(img.dtype) or len(img.shape) != 2:
 316                # Images of uint is not optimized by xxx_to_image
 317                self.img = self.cook_image(img.to_numpy())
 318            else:
 319                # Type matched! We can use an optimized copy kernel.
 320                assert img.shape == self.res, "Image resolution does not match GUI resolution"
 321                tensor_to_image(img, self.img)
 322                ti.sync()
 323
 324        elif isinstance(img, taichi.lang.matrix.MatrixField):
 325            if _ti_core.is_integral(img.dtype):
 326                self.img = self.cook_image(img.to_numpy())
 327            else:
 328                # Type matched! We can use an optimized copy kernel.
 329                assert img.shape == self.res, "Image resolution does not match GUI resolution"
 330                assert (
 331                    img.n in [2, 3, 4] and img.m == 1
 332                ), "Only greyscale, RG, RGB or RGBA images are supported in GUI.set_image"
 333
 334                vector_to_image(img, self.img)
 335                ti.sync()
 336
 337        elif isinstance(img, np.ndarray):
 338            self.img = self.cook_image(img)
 339
 340        else:
 341            raise ValueError(f"GUI.set_image only takes a Taichi field or NumPy array, not {type(img)}")
 342
 343        self.core.set_img(self.img.ctypes.data)

 345    def contour(self, scalar_field, normalize=False):
 346        """Plot a contour view of a scalar field.
 347
 348        The input scalar_field will be converted to a Numpy array first, and then plotted
 349        by the Matplotlib colormap 'Plasma'. Notice this method will automatically perform
 350        a bilinear interpolation on the field if the size of the field does not match with
 351        the GUI window size.
 352
 353        Args:
 354            scalar_field (ti.field): The scalar field being plotted.
 355            normalize (bool, Optional): Display the normalized scalar field if set to True.
 356            Default is False.
 357        """
 358        try:
 359            from matplotlib import cm  # pylint: disable=import-outside-toplevel
 360        except ImportError:
 361            raise RuntimeError(
 362                "Failed to import Matplotlib. Please install it via /\
 363            `pip install matplotlib` first. "
 364            )
 365        scalar_field_np = scalar_field.to_numpy()
 366        if self.res != scalar_field_np.shape:
 367            x, y = np.meshgrid(np.linspace(0, 1, self.res[1]), np.linspace(0, 1, self.res[0]))
 368            x_idx = x * (scalar_field_np.shape[1] - 1)
 369            y_idx = y * (scalar_field_np.shape[0] - 1)
 370            x1 = x_idx.astype(int)
 371            x2 = np.minimum(x1 + 1, scalar_field_np.shape[1] - 1)
 372            y1 = y_idx.astype(int)
 373            y2 = np.minimum(y1 + 1, scalar_field_np.shape[0] - 1)
 374            array_y1 = scalar_field_np[y1, x1] * (1 - (x_idx - x1)) * (1 - (y_idx - y1)) + scalar_field_np[y1, x2] * (
 375                x_idx - x1
 376            ) * (1 - (y_idx - y1))
 377            array_y2 = scalar_field_np[y2, x1] * (1 - (x_idx - x1)) * (y_idx - y1) + scalar_field_np[y2, x2] * (
 378                x_idx - x1
 379            ) * (y_idx - y1)
 380            output = array_y1 + array_y2
 381        else:
 382            output = scalar_field_np
 383        if normalize:
 384            scalar_max, scalar_min = np.max(output), np.min(output)
 385            output = (output - scalar_min) / (scalar_max - scalar_min)
 386        self.set_image(cm.plasma(output))

 398    def circles(self, pos, radius=1, color=0xFFFFFF, palette=None, palette_indices=None):
 399        """Draws a list of circles on canvas.
 400
 401        Args:
 402            pos (numpy.array): The positions of the circles.
 403            radius (Union[Number, numpy.array], optional): The radius of the circles in pixel. \
 404                Can be either a number, which will be applied to all circles, or a 1D NumPy array of the same length as `pos`. \
 405                The default is 1.
 406            color (int, optional): The color of the circles. Default is 0xFFFFFF.
 407            palette (list[int], optional): The List of colors from which to
 408                choose to draw. Default is None.
 409            palette_indices (Union[list[int], ti.field, numpy.array], optional):
 410                The List of indices that choose color from palette for each
 411                circle. Shape must match pos. Default is None.
 412
 413        """
 414        n = pos.shape[0]
 415        if len(pos.shape) == 3:
 416            assert pos.shape[2] == 1
 417            pos = pos[:, :, 0]
 418
 419        assert pos.shape == (n, 2)
 420        pos = np.ascontiguousarray(pos.astype(np.float32))
 421        # Note: do not use "pos = int(pos.ctypes.data)" here
 422        # Otherwise pos will get garbage collected by Python
 423        # and the pointer to its data becomes invalid
 424        pos_ptr = int(pos.ctypes.data)
 425
 426        if isinstance(color, np.ndarray):
 427            assert color.shape == (n,)
 428            color = np.ascontiguousarray(color.astype(np.uint32))
 429            color_array = int(color.ctypes.data)
 430            color_single = 0
 431        elif isinstance(color, int):
 432            color_array = 0
 433            color_single = color
 434        else:
 435            raise ValueError("Color must be an ndarray or int (e.g., 0x956333)")
 436
 437        if palette is not None:
 438            assert palette_indices is not None, "palette must be used together with palette_indices"
 439
 440            if isinstance(palette_indices, Field):
 441                ind_int = palette_indices.to_numpy().astype(np.uint32)
 442            elif isinstance(palette_indices, list) or isinstance(palette_indices, np.ndarray):
 443                ind_int = np.array(palette_indices).astype(np.uint32)
 444            else:
 445                try:
 446                    ind_int = np.array(palette_indices)
 447                except:
 448                    raise TypeError("palette_indices must be a type that can be converted to numpy.ndarray")
 449
 450            assert issubclass(ind_int.dtype.type, np.integer), "palette_indices must be an integer array"
 451            assert ind_int.shape == (n,), "palette_indices must be in 1-d shape with shape (num_particles, )"
 452            assert min(ind_int) >= 0, "the min of palette_indices must not be less than zero"
 453            assert max(ind_int) < len(palette), "the max of palette_indices must not exceed the length of palette"
 454            color_array = np.array(palette, dtype=np.uint32)[ind_int]
 455            color_array = np.ascontiguousarray(color_array)
 456            color_array = color_array.ctypes.data
 457
 458        if isinstance(radius, np.ndarray):
 459            assert radius.shape == (n,)
 460            radius = np.ascontiguousarray(radius.astype(np.float32))
 461            radius_array = int(radius.ctypes.data)
 462            radius_single = 0
 463        elif isinstance(radius, numbers.Number):
 464            radius_array = 0
 465            radius_single = radius
 466        else:
 467            raise ValueError("Radius must be an ndarray or float (e.g., 0.4)")
 468
 469        self.canvas.circles_batched(n, pos_ptr, color_single, color_array, radius_single, radius_array)

 601    @staticmethod
 602    def _arrow_to_lines(orig, major, tip_scale=0.2, angle=45):
 603        angle = math.radians(180 - angle)
 604        c, s = math.cos(angle), math.sin(angle)
 605        minor1 = np.array([major[:, 0] * c - major[:, 1] * s, major[:, 0] * s + major[:, 1] * c]).swapaxes(0, 1)
 606        minor2 = np.array([major[:, 0] * c + major[:, 1] * s, -major[:, 0] * s + major[:, 1] * c]).swapaxes(0, 1)
 607        end = orig + major
 608        return [
 609            (orig, end),
 610            (end, end + minor1 * tip_scale),
 611            (end, end + minor2 * tip_scale),

 720    def vector_field(self, vector_field, arrow_spacing=5, color=0xFFFFFF):
 721        """Display a vector field on canvas.
 722
 723        Args:
 724            vector_field (ti.Vector.field): The vector field being displayed.
 725            arrow_spacing (int, optional): The spacing between vectors.
 726            color (Union[int, np.array], optional): The color of vectors.
 727
 728        """
 729        v_np = vector_field.to_numpy()
 730        v_norm = np.linalg.norm(v_np, axis=-1)
 731        nx, ny, ndim = v_np.shape
 732        max_magnitude = np.max(v_norm)  # Find the largest vector magnitude
 733
 734        # The largest vector should occupy 10% of the window
 735        scale_factor = 0.1 / (max_magnitude + 1e-16)
 736
 737        x = np.arange(0, 1, arrow_spacing / nx)
 738        y = np.arange(0, 1, arrow_spacing / ny)
 739        X, Y = np.meshgrid(x, y)
 740        begin = np.dstack((X, Y)).reshape(-1, 2, order="F")
 741        incre = (v_np[::arrow_spacing, ::arrow_spacing] * scale_factor).reshape(-1, 2, order="C")
 742        self.arrows(orig=begin, direction=incre, radius=1, color=color)

 764        def __init__(self, *e_filter):

 825

 827            :class:`~taichi.misc.gui.GUI.EVENT`: The keyboard triggered event.

 831        e = GUI.Event()
 832        event = self.core.get_key_event_head()
 833
 834        e.type = event.type
 835        e.key = event.key
 836        e.pos = self.core.canvas_untransform(event.pos)
 837        e.pos = (e.pos[0], e.pos[1])

 839

 841            e.delta = event.delta
 842        else:
 843            e.delta = (0, 0)

 845        for mod in ["Shift", "Alt", "Control"]:
 846            if self.is_pressed(mod):
 847                e.modifier.append(mod)

 851        else:

228    def cook_image(self, img):
229        """Converts an img to range [0, 1] for display.
230
231        The input image is stored in a `numpy.ndarray`, if it's dtype
232        is `int` it will be rescaled and mapped into range [0, 1]. If
233        the dtype is `float` it will be directly casted to 32-bit float type.
234        """
235        if img.dtype in [np.uint8, np.uint16, np.uint32, np.uint64]:
236            img = img.astype(np.float32) * (1 / np.iinfo(img.dtype).max)
237        elif img.dtype in [np.float16, np.float32, np.float64]:
238            img = img.astype(np.float32)
239        else:
240            raise ValueError(f"Data type {img.dtype} not supported in GUI.set_image")
241
242        if len(img.shape) == 2:
243            img = img[..., None]
244
245        if img.shape[2] == 1:
246            img = img + np.zeros((1, 1, 4), np.float32)
247        if img.shape[2] == 3:
248            zeros = np.zeros((img.shape[0], img.shape[1], 1), np.float32)
249            img = np.concatenate([img, zeros], axis=2)
250        if img.shape[2] == 2:
251            zeros = np.zeros((img.shape[0], img.shape[1], 2), np.float32)
252            img = np.concatenate([img, zeros], axis=2)
253
254        assert img.shape[2] == 4, "Image must be grayscale, RG, RGB or RGBA"
255
256        res = img.shape[:2]
257        assert res == self.res, "Image resolution does not match GUI resolution"
258        return np.ascontiguousarray(img)

270    def set_image(self, img):
271        """Sets an image to display on the window.
272
273        The image pixels are set from the values of `img[i, j]`, where `i` indicates
274        the horizontal coordinates (from left to right) and `j` the vertical coordinates
275        (from bottom to top).
276
277        If the window size is `(x, y)`, then `img` must be one of:
278            - `ti.field(shape=(x, y))`, a gray-scale image
279            - `ti.field(shape=(x, y, 3))`, where `3` is for `(r, g, b)` channels
280            - `ti.field(shape=(x, y, 2))`, where `2` is for `(r, g)` channels
281            - `ti.Vector.field(3, shape=(x, y))` `(r, g, b)` channels on each component
282            - `ti.Vector.field(2, shape=(x, y))` `(r, g)` channels on each component
283            - `np.ndarray(shape=(x, y))`
284            - `np.ndarray(shape=(x, y, 3))`
285            - `np.ndarray(shape=(x, y, 2))`
286
287        The data type of `img` must be one of:
288            - `uint8`, range `[0, 255]`
289            - `uint16`, range `[0, 65535]`
290            - `uint32`, range `[0, 4294967295]`
291            - `float32`, range `[0, 1]`
292            - `float64`, range `[0, 1]`
293
294        Args:
295            img (Union[:class:`taichi.field`, `numpy.array`]): The color array \
296                representing the image to be drawn. Support greyscale, RG, RGB, \
297                and RGBA color representations. Its shape must match GUI resolution.
298        """
299
300        if self.fast_gui:
301            assert isinstance(
302                img, taichi.lang.matrix.MatrixField
303            ), "Only ti.Vector.field is supported in GUI.set_image when fast_gui=True"
304            assert img.shape == self.res, "Image resolution does not match GUI resolution"
305            assert img.n in [3, 4] and img.m == 1, "Only RGB images are supported in GUI.set_image when fast_gui=True"
306            assert img.dtype in [
307                ti.f32,
308                ti.f64,
309                ti.u8,
310            ], "Only f32, f64, u8 are supported in GUI.set_image when fast_gui=True"
311
312            vector_to_fast_image(img, self.img)
313            return
314
315        if isinstance(img, ScalarField):
316            if _ti_core.is_integral(img.dtype) or len(img.shape) != 2:
317                # Images of uint is not optimized by xxx_to_image
318                self.img = self.cook_image(img.to_numpy())
319            else:
320                # Type matched! We can use an optimized copy kernel.
321                assert img.shape == self.res, "Image resolution does not match GUI resolution"
322                tensor_to_image(img, self.img)
323                ti.sync()
324
325        elif isinstance(img, taichi.lang.matrix.MatrixField):
326            if _ti_core.is_integral(img.dtype):
327                self.img = self.cook_image(img.to_numpy())
328            else:
329                # Type matched! We can use an optimized copy kernel.
330                assert img.shape == self.res, "Image resolution does not match GUI resolution"
331                assert (
332                    img.n in [2, 3, 4] and img.m == 1
333                ), "Only greyscale, RG, RGB or RGBA images are supported in GUI.set_image"
334
335                vector_to_image(img, self.img)
336                ti.sync()
337
338        elif isinstance(img, np.ndarray):
339            self.img = self.cook_image(img)
340
341        else:
342            raise ValueError(f"GUI.set_image only takes a Taichi field or NumPy array, not {type(img)}")
343
344        self.core.set_img(self.img.ctypes.data)

346    def contour(self, scalar_field, normalize=False):
347        """Plot a contour view of a scalar field.
348
349        The input scalar_field will be converted to a Numpy array first, and then plotted
350        by the Matplotlib colormap 'Plasma'. Notice this method will automatically perform
351        a bilinear interpolation on the field if the size of the field does not match with
352        the GUI window size.
353
354        Args:
355            scalar_field (ti.field): The scalar field being plotted.
356            normalize (bool, Optional): Display the normalized scalar field if set to True.
357            Default is False.
358        """
359        try:
360            from matplotlib import cm  # pylint: disable=import-outside-toplevel
361        except ImportError:
362            raise RuntimeError(
363                "Failed to import Matplotlib. Please install it via /\
364            `pip install matplotlib` first. "
365            )
366        scalar_field_np = scalar_field.to_numpy()
367        if self.res != scalar_field_np.shape:
368            x, y = np.meshgrid(np.linspace(0, 1, self.res[1]), np.linspace(0, 1, self.res[0]))
369            x_idx = x * (scalar_field_np.shape[1] - 1)
370            y_idx = y * (scalar_field_np.shape[0] - 1)
371            x1 = x_idx.astype(int)
372            x2 = np.minimum(x1 + 1, scalar_field_np.shape[1] - 1)
373            y1 = y_idx.astype(int)
374            y2 = np.minimum(y1 + 1, scalar_field_np.shape[0] - 1)
375            array_y1 = scalar_field_np[y1, x1] * (1 - (x_idx - x1)) * (1 - (y_idx - y1)) + scalar_field_np[y1, x2] * (
376                x_idx - x1
377            ) * (1 - (y_idx - y1))
378            array_y2 = scalar_field_np[y2, x1] * (1 - (x_idx - x1)) * (y_idx - y1) + scalar_field_np[y2, x2] * (
379                x_idx - x1
380            ) * (y_idx - y1)
381            output = array_y1 + array_y2
382        else:
383            output = scalar_field_np
384        if normalize:
385            scalar_max, scalar_min = np.max(output), np.min(output)
386            output = (output - scalar_min) / (scalar_max - scalar_min)
387        self.set_image(cm.plasma(output))

602    @staticmethod
603    def _arrow_to_lines(orig, major, tip_scale=0.2, angle=45):
604        angle = math.radians(180 - angle)
605        c, s = math.cos(angle), math.sin(angle)
606        minor1 = np.array([major[:, 0] * c - major[:, 1] * s, major[:, 0] * s + major[:, 1] * c]).swapaxes(0, 1)
607        minor2 = np.array([major[:, 0] * c + major[:, 1] * s, -major[:, 0] * s + major[:, 1] * c]).swapaxes(0, 1)
608        end = orig + major
609        return [
610            (orig, end),
611            (end, end + minor1 * tip_scale),
612            (end, end + minor2 * tip_scale),

721    def vector_field(self, vector_field, arrow_spacing=5, color=0xFFFFFF):
722        """Display a vector field on canvas.
723
724        Args:
725            vector_field (ti.Vector.field): The vector field being displayed.
726            arrow_spacing (int, optional): The spacing between vectors.
727            color (Union[int, np.array], optional): The color of vectors.
728
729        """
730        v_np = vector_field.to_numpy()
731        v_norm = np.linalg.norm(v_np, axis=-1)
732        nx, ny, ndim = v_np.shape
733        max_magnitude = np.max(v_norm)  # Find the largest vector magnitude
734
735        # The largest vector should occupy 10% of the window
736        scale_factor = 0.1 / (max_magnitude + 1e-16)
737
738        x = np.arange(0, 1, arrow_spacing / nx)
739        y = np.arange(0, 1, arrow_spacing / ny)
740        X, Y = np.meshgrid(x, y)
741        begin = np.dstack((X, Y)).reshape(-1, 2, order="F")
742        incre = (v_np[::arrow_spacing, ::arrow_spacing] * scale_factor).reshape(-1, 2, order="C")
743        self.arrows(orig=begin, direction=incre, radius=1, color=color)
