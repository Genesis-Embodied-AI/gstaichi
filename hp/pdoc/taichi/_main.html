 29def timer(func):
 30    """Function decorator to benchmark a function running time."""
 31
 32    @wraps(func)
 33    def wrapper(*args, **kwargs):
 34        start = timeit.default_timer()
 35        result = func(*args, **kwargs)
 36        elapsed = timeit.default_timer() - start
 37        print(f">>> Running time: {elapsed:.2f}s")
 38        return result
 39
 40    return wrapper

 43def registerableCLI(cls):
 44    """Class decorator to register methods with @register into a set."""
 45    cls.registered_commands = set([])
 46    for name in dir(cls):
 47        method = getattr(cls, name)
 48        if hasattr(method, "registered"):
 49            cls.registered_commands.add(name)
 50    return cls

633                return "::".join(reversed(file[10:-4].split("__arch_")))
